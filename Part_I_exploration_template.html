# function to plot graph
def Visualize(feature, title):
    """ Visualize distribution of the feature"""
    #select a color to display
    color = sb.color_palette()[0]
    #ordering bars
    desc = biketrip[feature].value_counts().index
    #barchart
    sb.countplot(data=biketrip, x=feature, color=color, order=desc)
    plt.title(title);
    #calculate percentage
    total = biketrip[feature].value_counts().sum()
    unit = biketrip[feature].value_counts()
    #generate label and ticks
    locs,labels = plt.xticks()
    ticks = zip(locs,labels)
    # to display percentage of the different categories
    for loc, label in ticks:
        x = unit[label.get_text()]
        percent = '{:0.1f}%'.format(100*x/total)
        plt.text(loc,x+2,percent, ha='center',color='black')
